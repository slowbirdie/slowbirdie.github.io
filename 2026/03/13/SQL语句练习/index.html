    查询a表中男生所有的信息

    select 学号,姓名,性别,出生日期,专业成绩 from a where 性别=”男’’;

    或

    select * from a where 性别=’’男’’;

    查询a表中男生的姓名和专业成绩

    select 姓名,专业成绩 from a where 性别=’’男’’;

    查询a表中男生的所有信息并按专业成绩升序排序

    select * from a where 性别=’’男’’ order by 专业成绩;

    查询a表中男生的姓名和专业成绩,并将查询结果输出到“男生成绩”表

    select 性别,专业成绩 into 男生成绩 from a where 性别=’’男”;

    按性别统计性别的平均专业成绩

    select 性别,avg(专业成绩) as 平均专业成绩 from a group by 性别;

    统计男生的平均专业成绩

    select 性别,avg(专业成绩) as 平均专业成绩 from a group by 性别 having 性别=’’男’’;

    将a表中学号为“20170105”的学生姓名更改为“孙韬”

    update a set 姓名=”孙韬’’ where 学号=’’20170105’’;

    在a表中插入一条记录：学号：20170106 性别：女 出生日期：2000/09/13 专业成绩：95

    insert into a (学号,姓名,性别,出生日期,专业成绩) values(‘’20170106’’,’’刘芸’’,女,#2000/9/13#,95);

    或

    Insert into a values(‘’20170106’’,’’刘芸’’,女,#2000/9/13#,95);

    删除a表中学号为“20170101”的记录

    drop from a where 学号=’’20170101’’;

    查询被一个学生选修的课程有几个?（count）

    select count(Cno) from SC where Sno=’…’;

    查询姓名为张三的学生的选修课程有多少?

    select count(Cno) from SC,S where Sname=’张三’ and SC.Sno=S.Sno;

    或

    select count(Cno) from SC where Sno in (select Sno from S where Sname=’张三’);

    查询年龄在18~19之间的学生信息（between）

    select * from S where age between 18 and 19;

    查询姓刘的同学的信息（like）

    select * from S where Sname like ‘刘%’;

    查询姓刘且姓名仅仅两字的信息

    select * from S where Sname like ‘刘_’;

    查询没有先修课的课程信息（is null）

    select * from C where Cpno is null;

    求各门课学生的平均成绩,显示课程号及相应的平均成绩（avg,group by）

    select Cno,avg(grade) from SC group by Cno;

    查询有2个以上学生选修的课程的课程号（having）

    select Cno from SC group by Cno having count(Sno)>2;

    查询学生选修课程的情况,要求显示学生名,课程名,成绩（三表查询）

    select Sname,Cname,grade from S,C,SC where S.Sno=SC.Sno and C.Cno=SC.Cno;

    查询选修了全部课程的学生姓名

    select Sname from S where not exists(select from C where not exists(select * from SC where S.Sno=SC.Sno and C.Cno=SC.Cno));

    或

    select Sname from S join SC on S.Sno = SC.Sno group by S.Sno, Sname having count(distinc Cno) = (select count(*) from C);

    或

    select Sname from S where Sno = all(select Sno from SC where SC.Cno = C.Cno);

    创建dptg_age,包括两个字段：系别和平均年龄。按系分组求平均年龄,把系名和平均年龄

    存入新表中,一步完成。

    create table dept_age as select department as 系别,avg(age) as 平均年龄 from student group by department；

    将女生的年龄减去1岁

    update student set age = age − 1 where sex = ‘女’;

    将信息科学系的全体学生成绩设为100分

    update SC set grade = 100 where Sno in (select Sno from student where dept = ‘信息科学系’)

    删除200215121学生的选课信息

    delete from SC where Sno = ‘200215121’;

    将成绩表中计算机科学系的学生信息删除

    delete from SC where Sno in (select Sno from Student where Sdept = ‘计算机科学系’);

    建立视图view_grade,内容为学生姓名、课程名、成绩

    create view view_grade as select student.Sname as 学生姓名,source.Cname as 课程名,SC.grade as 成绩 from student join SC on student.Sno = SC.Sno join cource on SC.Cno = cource.Cno);

    或

    create view view_grade as select student.Sname as 学生姓名,cource.Cname as 课程名,SC.grade as 成绩 from student,cource,sc where student.Sno = SC.Sno and cource.Cno = SC.Cno;

    查询课程表的所有信息

    SELECT * FROM 课程表;

    查询课程名及其学分

    SELECT 课程名, 学分 FROM 课程表;

    查询学生姓名及其出生年份,出生年份的别名为“出生年份”

    SELECT 姓名, 出生年份 AS 出生年份 FROM 学生表;

    查询被学生选修的课程的课程名

    SELECT DISTINCT 课程名 FROM 课程表 WHERE 课程号 IN (SELECT DISTINCT 课程号 FROM 成绩表);

    查询考试需要补考的学生有几个？(distinct）

    SELECT COUNT(DISTINCT 学号) FROM 成绩表 WHERE 成绩 < 60;

    查询每门课的间接先修课，要求显示课程名及该课程间接先修课的课程名。（三表查询）

    SELECT a.课程名, c.课程名 AS 间接先修课名 FROM 课程表 a LEFT JOIN 课程表 b ON a.先修课号 = b.课程号 LEFT JOIN 课程表 c ON b.先修课号 = c.课程号 WHERE c.课程名 IS NOT NULL;

    查询成绩在90分以上的女生（复合条件）

    SELECT DISTINCT 学生表.* FROM 学生表 JOIN 成绩表 ON 学生表.学号 = 成绩表.学号 WHERE 成绩表.成绩 > 90 AND 学生表.性别 = ‘女’;

    查询与王敏在一个系学习的学生 （in）

    SELECT * FROM 学生表 WHERE 系别 = (SELECT 系别 FROM 学生表 WHERE 姓名 = ‘王敏’) AND 姓名 != ‘王敏’;

    查询其他系中比信息系某一学生年龄大的学生姓名和年龄

    SELECT 姓名, 年龄 FROM 学生表 WHERE 系别 != ‘信息系’ AND 年龄 > ANY (SELECT 年龄 FROM 学生表 WHERE 系别 = ‘信息系’);

    查询其他系中比信息系所有学生年龄大的学生姓名和年龄

    SELECT 姓名, 年龄 FROM 学生表 WHERE 系别 != ‘信息系’ AND 年龄 > ALL (SELECT 年龄 FROM 学生表 WHERE 系别 = ‘信息系’);

    查询被所有学生选修的课程名

    SELECT 课程名 FROM 课程表 WHERE 课程号 IN (SELECT 课程号 FROM 成绩表 GROUP BY 课程号 HAVING COUNT(DISTINCT 学号) = (SELECT COUNT(*) FROM 学生表));

    查询选修了学生95002选修的全部课程的学生学号

    SELECT 学号 FROM 学生表 WHERE 学号 != ‘95002’ AND NOT EXISTS (SELECT 1 FROM 成绩表 c WHERE c.学号 = ‘95002’ AND NOT EXISTS (SELECT 1 FROM 成绩表 d WHERE d.学号 = 学生表.学号 AND d.课程号 = c.课程号));

    查询学生成绩5-7名的学生学号、课程号、成绩

    SELECT 学号, 课程号, 成绩 FROM (SELECT 学号, 课程号, 成绩, ROW_NUMBER() OVER (ORDER BY 成绩 DESC) AS rn FROM 成绩表) t WHERE rn BETWEEN 5 AND 7;

    分别对学生表、成绩表进行：内连接、左外连接、右外连接、全连接

    内连接：SELECT * FROM 学生表 INNER JOIN 成绩表 ON 学生表.学号 = 成绩表.学号;

    左外连接：SELECT * FROM 学生表 LEFT JOIN 成绩表 ON 学生表.学号 = 成绩表.学号;

    右外连接：SELECT * FROM 学生表 RIGHT JOIN 成绩表 ON 学生表.学号 = 成绩表.学号;

    全连接：SELECT * FROM 学生表 FULL OUTER JOIN 成绩表 ON 学生表.学号 = 成绩表.学号;

    找出每个学生低于他选修课程平均成绩的课程号

    SELECT 学号, 课程号 FROM 成绩表 a WHERE 成绩 < (SELECT AVG(成绩) FROM 成绩表 b WHERE b.学号 = a.学号);
